package com.demo;

import java.io.IOException;
import java.io.PrintWriter;
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;

import javax.servlet.ServletException;
import javax.servlet.annotation.WebServlet;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;


@WebServlet("/save")
public class SaveData extends HttpServlet {

	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		
		String id = request.getParameter("id");
		String name = request.getParameter("name");
		String age = request.getParameter("age");
		String email = request.getParameter("email");
		String password = request.getParameter("pwd");
		
		try {
			Class.forName("com.mysql.cj.jdbc.Driver");
			Connection conn = DriverManager.getConnection("jdbc:mysql://localhost:3306/aaaa", "root","Allusai@123");
			conn.setAutoCommit(false);
			PreparedStatement psmt = conn.prepareStatement("insert into Doctor values(?,?,?,?,?)");
			psmt.setInt(1, Integer.parseInt(id));
			psmt.setString(2, name);
			psmt.setInt(3, Integer.parseInt(age));
			psmt.setString(4, email);
			psmt.setString(5, password);
			
			int status = psmt.executeUpdate();
			if(status == 1) 
			{
				conn.commit();
				System.out.println("transaction successful..");
			}
			else {
				conn.rollback();
				System.out.println("transaction is failed...");
			}
			
		} catch (ClassNotFoundException | SQLException e) {
			e.printStackTrace();
		}
	}
}
